src/lib/maps/places.ts — fix paginação robusta + logging
export async function searchPlaces(
  query: string, location: string,
  radius: number = 5000, maxPages: number = 3
) {
  if (!GOOGLE_MAPS_API_KEY) throw new Error("GOOGLE_MAPS_API_KEY não configurada.");

  let allResults: any[] = [];
  let nextToken: string | null = null;
  let pagesFetched = 0;
  let consecutiveErrors = 0;

  do {
    try {
      // Token precisa de 2-3s para ativar — aumentar para 3s para ser seguro
      if (nextToken) await new Promise(r => setTimeout(r, 3000));

      const url = nextToken
        ? `https://maps.googleapis.com/maps/api/place/textsearch/json?pagetoken=${nextToken}&key=${GOOGLE_MAPS_API_KEY}`
        : `https://maps.googleapis.com/maps/api/place/textsearch/json?query=${encodeURIComponent(query)}&location=${location}&radius=${radius}&key=${GOOGLE_MAPS_API_KEY}&language=pt-BR`;

      const { data } = await axios.get(url, { timeout: 15000 });
      const { results = [], next_page_token, status } = data;

      if (status === "INVALID_REQUEST" && nextToken) {
        // Token ainda não ativo — retry com delay maior
        consecutiveErrors++;
        if (consecutiveErrors >= 2) break; // desistir após 2 retries
        await new Promise(r => setTimeout(r, 3000));
        continue;
      }

      if (status !== "OK" && status !== "ZERO_RESULTS") {
        console.error("Places API status:", status);
        break;
      }

      consecutiveErrors = 0;
      allResults = [...allResults, ...results];
      nextToken = next_page_token || null;
      pagesFetched++;

      console.log(`[Places] Página ${pagesFetched}: +${results.length} (total: ${allResults.length})`);

    } catch (err) {
      console.error("Places page error:", err);
      consecutiveErrors++;
      if (consecutiveErrors >= 2) break;
    }

  } while (nextToken && pagesFetched < maxPages);

  console.log(`[Places] Total: ${allResults.length} resultados em ${pagesFetched} páginas`);
  return allResults.map((p: any) => ({
    name: p.name, address: p.formatted_address, place_id: p.place_id,
    rating: p.rating, user_ratings_total: p.user_ratings_total,
    lat: p.geometry.location.lat, lng: p.geometry.location.lng,
    types: p.types, business_status: p.business_status,
  }));
}

import { sendWhatsAppMessage } from "@/lib/whatsapp/evolution";
import { sendEmail } from "@/lib/emails/resend";

export async function sendCampaignMessage(leadId: string, campaignId: string) {
  const supabase = await createClient();
  
  // Buscar dados completos
  const { data: lead } = await supabase
    .from("leads")
    .select("*, campaigns(*, businesses(*, agents(config), metadata))")
    .eq("id", leadId).single();

  const campaign = lead?.campaigns;
  const business = campaign?.businesses;
  const integrations = business?.metadata?.integrations || {};

  // Gerar mensagem com IA
  const message = await generateOutreachMessage(
    lead, business, campaign, business.agents?.[0]?.config
  );

  // Enviar pelo canal correto
  if (campaign.channel === 'whatsapp' && lead.phone) {
    const ev = integrations.evolution;
    if (ev?.url && ev?.key && ev?.instance) {
      await sendWhatsAppMessage({
        to: lead.phone, message,
        evolutionUrl: ev.url, apiKey: ev.key, instance: ev.instance,
      });
    }
  } else if (campaign.channel === 'email') {
    const resendKey = integrations.resend?.key;
    if (resendKey && lead.email) {
      await sendEmail({ to: lead.email, subject: "...", body: message, resendKey });
    }
  }

  // Salvar mensagem e atualizar status
  await supabase.from("leads")
    .update({ 
      status: 'contacted',
      metadata: { ...lead.metadata, last_message: message, contacted_at: new Date().toISOString() }
    })
    .eq("id", leadId);
}

    setNicheInput('');
  }
};

// No JSX:
<div className="space-y-2">
  <Label>Nichos para buscar (até 5)</Label>
  <div className="flex gap-2">
    <Input
      value={nicheInput}
      onChange={e => setNicheInput(e.target.value)}
      onKeyDown={e => e.key === 'Enter' && addNiche()}
      placeholder="Ex: Restaurantes, Academias..."
    />
    <Button type="button" onClick={addNiche}>+</Button>
  </div>
  <div className="flex flex-wrap gap-2 mt-2">
    {niches.map(n => (
      <span key={n} className="badge">
        {n}
        <button onClick={() => setNiches(prev => prev.filter(x => x !== n))}>×</button>
      </span>
    ))}
  </div>
</div>

// No submit, chamar searchLeadsAction para cada nicho em sequência:
for (const niche of niches) {
  await searchLeadsAction(niche, region, campaignId);
}
